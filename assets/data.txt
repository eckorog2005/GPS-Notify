CREATE TABLE Location(_id INTEGER PRIMARY KEY AUTOINCREMENT, locationName VARCHAR(80), latitude DOUBLE, longitude DOUBLE);
CREATE TABLE ScheduledLocationTask(_id INTEGER PRIMARY KEY AUTOINCREMENT, taskID INTEGER, locationID INTEGER, occurrence VARCHAR(10), note VARCHAR(140), lastUsed Long, alartReady VARCHAR(6), FOREIGN KEY(taskID) REFERENCES Task(_id), FOREIGN KEY(locationID) REFERENCES Location(_id));
CREATE TABLE ScheduledTimeTask(_id INTEGER PRIMARY KEY AUTOINCREMENT, taskID INTEGER, time Long, occurrence SHORT, note VARCHAR(140), lastUsed Long, FOREIGN KEY(taskID) REFERENCES Task(_id));
CREATE TABLE Task(_id INTEGER PRIMARY KEY AUTOINCREMENT, taskName VARCHAR(80), numOfUses LONG);
